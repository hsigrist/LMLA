\begin{frame}
  \frametitle{Transversal de gravedad}
  \begin{block}{Definición}
    Considere el $\triangle ABC$ de la Figura \ref{fig:tg}. Una transversal de gravedad es un trazo de
    recta dibujado desde un vértice al punto medio del lado opuesto.

    \begin{figure}[h]
      \centering
      \begin{tikzpicture}[scale=.5]
        \tkzDefPoint(-1,1){A}
        \tkzDefPoint(5,1){B}
        \tkzDefEquilateral(A,B)\tkzGetPoint{C}
        \tkzDrawPolygon(A,B,C)
        \tkzCentroid(A,B,C)\tkzGetPoint{G}
        \tkzDrawPoint(G)
        \tkzDrawLines[add = 0 and 2/3](A,G B,G C,G)
        \tkzLabelPoint[left](A){$A$}
        \tkzLabelPoint[right](B){$B$}
        \tkzLabelPoint[above](C){$C$}
        \tkzLabelLine[pos=1.8,top](A,G){$t_A$}
        \tkzLabelLine[pos=1.8,top](B,G){$t_B$}
        \tkzLabelLine[pos=1.8,top](C,G){$t_C$}
     \end{tikzpicture}
      \caption{Triángulo $ABC$}
      \label{fig:tg}
    \end{figure}
  \end{block}
\end{frame}

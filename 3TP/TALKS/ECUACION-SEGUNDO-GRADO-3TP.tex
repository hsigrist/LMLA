      \label{eq:1}
      ax^2+bx+c&=&0
    \end{eqnarray}
    \begin{description}
    \item[$a$] coeficiente cuadrático (grado $2$)
    \item[$b$] coeficiente lineal (grado $1$)
    \item[$c$] coeficiente constante (grado $0$)
    \end{description}
  \end{block}
\end{frame}
